Nginx的安装和使用

Written by yyc on November 15, 2019 in Linux

Nginx的安装

本文在centos7环境下，使用源码编译和安装nginx。

在安装·nginx之前，首先安装一些必要的依赖：

yum update
yum -y install pcre pcre-devel zlib zlib-devel openssl openssl-devel

到官网下载页面选择一个稳定的nginx版本，下载、解压：

wget https://nginx.org/download/nginx-1.16.1.tar.gz
tar -zxvf nginx-1.16.1.tar.gz

配置、编译、安装：

cd nginx-1.16.1
./configure \
--prefix=/usr/local/nginx \
--with-http_ssl_module \
--with-http_stub_status_module \
--with-http_realip_module \
--with-threads
make
make install

设置软链接：

ln -s /usr/local/nginx/sbin/nginx /usr/bin/nginx

测试：

curl localhost

Nginx的配置

  • 在/var下创建文件夹www用于存放各个网站内容
  • 在/usr/local/nginx下创建一个sites-enabled的目录，用于存放各个网站的配置信息。

在1.16.1版本下，nginx的默认配置文件位置是：/usr/local/nginx/conf/nginx.conf，，

比如，我现在想创建一个域名为mall.com的网站，那么应该存在的目录结构为：

cd /var/www/mall.com
cd /usr/local/nginx/sites-enabled/mall.com.conf

mall.com的内容基本上就是一个静态网站，根目录包含index.html网页之类。

mall.com.conf的格式如下：

server {
	listen 443 ssl http2;
	listen [::]:443 ssl http2;

	server_name mall.com;
	root /var/www/mall.com/public;

	# SSL
	ssl_certificate /etc/letsencrypt/live/mall.com/fullchain.pem;
	ssl_certificate_key /etc/letsencrypt/live/mall.com/privkey.pem;
	ssl_trusted_certificate /etc/letsencrypt/live/mall.com/chain.pem;

	# security
	include nginxconfig.io/security.conf;

	# index.html fallback
	location / {
		try_files $uri $uri/ /index.html;
	}

	# additional config
	include nginxconfig.io/general.conf;
}

# HTTP redirect
server {
	listen 80;
	listen [::]:80;

	server_name mall.com;

	include nginxconfig.io/letsencrypt.conf;

	location / {
		return 301 https://mall.com$request_uri;
	}
}

在默认的配置文件末尾加上：

user www-data;
pid /run/nginx.pid;
worker_processes auto;
worker_rlimit_nofile 65535;

events {
	multi_accept on;
	worker_connections 65535;
}

http {
	charset utf-8;
	sendfile on;
	tcp_nopush on;
	tcp_nodelay on;
	server_tokens off;
	log_not_found off;
	types_hash_max_size 2048;
	client_max_body_size 16M;

	# MIME
	include mime.types;
	default_type application/octet-stream;

	# logging
	access_log /var/log/nginx/access.log;
	error_log /var/log/nginx/error.log warn;

	# SSL
	ssl_session_timeout 1d;
	ssl_session_cache shared:SSL:10m;
	ssl_session_tickets off;

	# Diffie-Hellman parameter for DHE ciphersuites
	ssl_dhparam /etc/nginx/dhparam.pem;

	# Mozilla Intermediate configuration
	ssl_protocols TLSv1.2 TLSv1.3;
	ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305:DHE-RSA-AES128-GCM-SHA256:DHE-RSA-AES256-GCM-SHA384;

	# OCSP Stapling
	ssl_stapling on;
	ssl_stapling_verify on;
	resolver 1.1.1.1 1.0.0.1 8.8.8.8 8.8.4.4 208.67.222.222 208.67.220.220 valid=60s;
	resolver_timeout 2s;

	# load configs
	include ../sites-enabled/*;
}
